title: 'Note Template'

This document outlines the main structures for creating **Note Template** objects.

## NoteTemplate

<ParamField name="NoteTemplate" type="Object">
  <Expandable title="properties">
    <ParamField name="globalPrompt" type="string">
      A global prompt applying to the entire template.
    </ParamField>
    <ParamField name="id" type="string" required>
      A unique ID for the template.
    </ParamField>
    <ParamField name="sections" type="Array of Section objects" required>
      Ordered list of sections that make up the template.
    </ParamField>
    <ParamField name="title" type="string" required>
      Title of the template.
    </ParamField>
  </Expandable>
</ParamField>

<CodeGroup>
```json Example
{
  "id": "template-abc",
  "title": "Project Documentation Template",
  "globalPrompt": "Overall style: straightforward",
  "sections": [
    {
      "id": "heading-1",
      "type": "heading",
      "properties": {
        "level": 1,
        "text": "Introduction"
      },
      "children": []
    },
    {
      "id": "text-1",
      "type": "text",
      "prompt": "Explain the project background",
      "properties": {
        "detail_level": "medium",
        "tone": "formal"
      }
    },
    {
      "id": "list-1",
      "type": "list",
      "prompt": "List key objectives",
      "properties": {
        "list_type": "bullet",
        "min_items": 1,
        "max_items": 3
      }
    }
  ]
}
```
</CodeGroup>

## Section

<ParamField name="Section" type="Discriminated Union">
  A discriminated union that represents any one of the major section types. The <code>type</code> field 
  determines which schema applies.
  
  <Expandable title="possible types">
    <ParamField name="HeadingSection" type="Object">
      When <code>type = "heading"</code>
    </ParamField>
    <ParamField name="ListSection" type="Object">
      When <code>type = "list"</code>
    </ParamField>
    <ParamField name="TextSection" type="Object">
      When <code>type = "text"</code>
    </ParamField>
  </Expandable>
</ParamField>

<CodeGroup>
```json Example
[
  {
    "id": "heading-1",
    "type": "heading",
    "properties": { "level": 1, "text": "Overview" },
    "children": []
  },
  {
    "id": "text-1",
    "type": "text",
    "prompt": "Brief overview text",
    "properties": { "tone": "technical" }
  },
  {
    "id": "list-1",
    "type": "list",
    "prompt": "Key tasks",
    "properties": { "list_type": "bullet" }
  }
]
```
</CodeGroup>

## HeadingSection

**Important Note on Nesting**:  
There is a maximum heading nesting depth of **2**.

<ParamField name="HeadingSection" type="Object">
  <Expandable title="properties">
    <ParamField name="children" type="Array of Section objects" required>
      Nested sections under this heading.
    </ParamField>
    <ParamField name="id" type="string" required>
      Unique identifier for this section.
    </ParamField>
    <ParamField name="properties" type="Object" required>
      <Expandable title="properties">
        <ParamField name="bold" type="boolean">
          Whether the content is bold.
        </ParamField>
        <ParamField name="hideIfEmpty" type="boolean">
          Hide this heading if empty.
        </ParamField>
        <ParamField name="italic" type="boolean">
          Whether the content is italic.
        </ParamField>
        <ParamField name="level" type="number" required>
          Heading level (0 ≤ level ≤ 6).
        </ParamField>
        <ParamField name="text" type="string" required>
          The heading text.
        </ParamField>
      </Expandable>
    </ParamField>
    <ParamField name="result" type="string">
      Resulting text if generated or stored.
    </ParamField>
    <ParamField name="type" type="string" required>
      Must be set to "heading".
    </ParamField>
  </Expandable>
</ParamField>

<CodeGroup>
```json Example
{
  "id": "heading-1",
  "type": "heading",
  "properties": {
    "bold": true,
    "italic": false,
    "hideIfEmpty": false,
    "level": 1,
    "text": "Project Introduction"
  },
  "children": []
}
```
</CodeGroup>

## TextSection

<ParamField name="TextSection" type="Object">
  <Expandable title="properties">
    <ParamField name="id" type="string" required>
      Unique identifier for this section.
    </ParamField>
    <ParamField name="prompt" type="string" required>
      A prompt or instruction for generating/filling text.
    </ParamField>
    <ParamField name="properties" type="Object" required>
      <Expandable title="properties">
        <ParamField name="bold" type="boolean">
          Whether the content is bold.
        </ParamField>
        <ParamField name="italic" type="boolean">
          Whether the content is italic.
        </ParamField>
        <ParamField name="hideIfEmpty" type="boolean">
          Hide this text if empty.
        </ParamField>
        <ParamField name="formatting_style" type="string" default="markdown">
          Content formatting style. One of: "markdown", "plain".
        </ParamField>
        <ParamField name="detail_level" type="string" default="standard">
          Detail depth of text. One of: "exhaustive", "detailed", "standard", "brief", "minimal".
        </ParamField>
        <ParamField name="tone" type="string" default="formal">
          Tone of the text. One of: "formal", "casual", "technical", "friendly", "instructional".
        </ParamField>
        <ParamField name="emptyPlaceholder" type="string">
          Placeholder if the content is empty.
        </ParamField>
        <ParamField name="label" type="string">
          A label or title for the text.
        </ParamField>
        <ParamField name="bold_label" type="boolean">
          Whether the label itself is bold.
        </ParamField>
        <ParamField name="italic_label" type="boolean">
          Whether the label itself is italic.
        </ParamField>
        <ParamField name="list_type" type="string">
          If this text is displayed as a list item. One of: "bullet", "numeric".
        </ParamField>
        <ParamField name="max_paragraphs" type="number">
          Maximum paragraphs (if auto-generating).
        </ParamField>
        <ParamField name="max_sentences" type="number">
          Maximum sentences (if auto-generating).
        </ParamField>
        <ParamField name="min_paragraphs" type="number">
          Minimum paragraphs (if auto-generating).
        </ParamField>
        <ParamField name="min_sentences" type="number">
          Minimum sentences (if auto-generating).
        </ParamField>
      </Expandable>
    </ParamField>
    <ParamField name="result" type="string">
      Final or generated text.
    </ParamField>
    <ParamField name="type" type="string" required>
      Must be set to "text".
    </ParamField>
  </Expandable>
</ParamField>

<CodeGroup>
```json Example
{
  "id": "text-1",
  "type": "text",
  "prompt": "Provide a concise summary of the project scope.",
  "properties": {
    "bold": false,
    "italic": false,
    "hideIfEmpty": true,
    "formatting_style": "markdown",
    "detail_level": "brief",
    "tone": "casual",
    "label": "Project Scope",
    "bold_label": false,
    "italic_label": false,
    "min_sentences": 2,
    "max_sentences": 3,
    "emptyPlaceholder": "No scope defined yet"
  }
}
```
</CodeGroup>

## ListSection

<ParamField name="ListSection" type="Object">
  <Expandable title="properties">
    <ParamField name="id" type="string" required>
      Unique identifier for this section.
    </ParamField>
    <ParamField name="prompt" type="string" required>
      A prompt or instruction for generating/filling the list.
    </ParamField>
    <ParamField name="properties" type="Object" required>
      <Expandable title="properties">
        <ParamField name="bold" type="boolean">
          Whether the content is bold.
        </ParamField>
        <ParamField name="italic" type="boolean">
          Whether the content is italic.
        </ParamField>
        <ParamField name="hideIfEmpty" type="boolean">
          Hide this list if empty.
        </ParamField>
        <ParamField name="formatting_style" type="string" default="markdown">
          Content formatting style. One of: "markdown", "plain".
        </ParamField>
        <ParamField name="detail_level" type="string" default="standard">
          Detail depth for list items. One of: "exhaustive", "detailed", "standard", "brief", "minimal".
        </ParamField>
        <ParamField name="tone" type="string" default="formal">
          Tone of the list content. One of: "formal", "casual", "technical", "friendly", "instructional".
        </ParamField>
        <ParamField name="emptyPlaceholder" type="string">
          Placeholder if the list is empty.
        </ParamField>
        <ParamField name="list_type" type="string" required>
          The format of the list. One of: "bullet", "numeric".
        </ParamField>
        <ParamField name="max_items" type="number">
          Maximum number of items.
        </ParamField>
        <ParamField name="min_items" type="number">
          Minimum number of items.
        </ParamField>
      </Expandable>
    </ParamField>
    <ParamField name="result" type="Array of strings">
      The final or generated list items.
    </ParamField>
    <ParamField name="type" type="string" required>
      Must be set to "list".
    </ParamField>
  </Expandable>
</ParamField>

<CodeGroup>
```json Example
{
  "id": "list-1",
  "type": "list",
  "prompt": "List the main deliverables of the project.",
  "properties": {
    "bold": false,
    "italic": false,
    "hideIfEmpty": false,
    "formatting_style": "markdown",
    "detail_level": "minimal",
    "tone": "technical",
    "list_type": "bullet",
    "min_items": 1,
    "max_items": 5,
    "emptyPlaceholder": "No deliverables defined"
  }
}
```
</CodeGroup>
